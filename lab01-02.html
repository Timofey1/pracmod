def create_d(length):
    D = []
    for i in range(length):
        x1,x2 = np.random.uniform(0,a),np.random.uniform(0,a)
        y1,y2 = np.random.uniform(0,b),np.random.uniform(0,b)
        D.append(dist(x1,y1,x2,y2))
    D.sort()
    return D
